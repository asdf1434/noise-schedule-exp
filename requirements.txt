# [cuda12] pulls in matching nvidia-cudnn-cu12/nvidia-cublas-cu12 etc. wheels
# for jaxlib's CUDA plugin. Without this extra, jaxlib still finds a
# jax-cuda12-plugin install (if one's present) but falls back to whatever
# cuDNN happens to be on the node's system path -- on this cluster that's an
# old 9.1.0, while jaxlib expects >=9.8.0, causing a hard-to-diagnose
# "DNN library initialization failed" crash at the first real jax op that's
# identical across every node (not node-specific flakiness).
jax[cuda12]
equinox
optax
pillow
importlib_resources
tensorflow-cpu
tensorflow-datasets
datasets
matplotlib
beartype
requests
certifi
tqdm

# clean-fid pulls in torch/torchvision unpinned, which resolves to the latest
# build (e.g. cu130) -- too new for the cluster's GPU drivers (max CUDA 12.9
# as of 2026-07-16, confirmed via evaluate_fid.py's GPU sweep failing with
# "CUDA initialization: driver too old" on 2.13.0+cu130). Pin to a cu121
# build instead: broadly compatible with drivers back to 525.60.13, well
# under the cluster's floor.
--extra-index-url https://download.pytorch.org/whl/cu121
torch==2.4.1+cu121
torchvision==0.19.1+cu121
clean-fid
